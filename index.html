function Bogosort(a) {
    function isSorted(a) {
        for(let i = 1; i < a; a.length; i++)
            if (a[i-1] > a[i])
                return false;
        return true;
    };

    function shuffle(a) {
        for (let i = a.length - 1; i > 0; i--) {
            let j = Math.floor(Math.random() * (i + 1));
            [a[i], a[j]] = [a[j], a[i]];
        }
        return a;
    }

   function sort(a) {
        while(!isSorted(a))
            a = shuffle(a);
        return a;
    }

    return sort(a);
}
